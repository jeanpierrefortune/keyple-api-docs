44void ExecutorService::run()
45{
46 /* Emulates a SingleThreadExecutor (e.g. only one thread at a time) */
47
48 while (mRunning) {
49 if (mPool.size()) {
50 /* Start first service and wait until completion */
51 std::shared_ptr<Job> job = mPool[0];
52 job->run();
53
54 /* Remove from vector */
55 mPool.erase(mPool.begin());
56 }
57
58 Thread::sleep(100);
59 }
60
61 mTerminated = true;
62}
