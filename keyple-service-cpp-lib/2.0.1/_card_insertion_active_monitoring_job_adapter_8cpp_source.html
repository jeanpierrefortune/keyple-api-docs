39void CardInsertionActiveMonitoringJobAdapter::CardInsertionActiveMonitoringJob::execute()
40{
41 try {
42 mParent->mLogger->debug("[%] Polling from isCardPresent\n", mParent->mReader->getName());
43
44 /* Re-init loop value to true */
45 mParent->mLoop = true;
46
47 while (mParent->mLoop) {
48 /* Polls for CARD_INSERTED */
49 if (mParent->mMonitorInsertion && mParent->mReader->isCardPresent()) {
50 mParent->mLogger->debug("[%] The card is present\n", mParent->mReader->getName());
51 mMonitoringState->onEvent(InternalEvent::CARD_INSERTED);
52 return;
53 }
54
55 /* Polls for CARD_REMOVED */
56 if (!mParent->mMonitorInsertion && !mParent->mReader->isCardPresent()) {
57 mParent->mLogger->debug("[%] The card is not present\n",
58 mParent->mReader->getName());
59 mParent->mLoop = false;
60 mMonitoringState->onEvent(InternalEvent::CARD_REMOVED);
61 return;
62 }
63
64 mRetries++;
65
66 mParent->mLogger->trace("[%] isCardPresent polling retries : %\n",
67 mParent->mReader->getName(),
68 mRetries);
69
70 try {
71 /* Wait a bit */
72 Thread::sleep(mParent->mCycleDurationMillis);
73 } catch (const InterruptedException& ignored) {
74 (void)ignored;
75 /* Restore interrupted state... */
76 interrupt();
77 mParent->mLoop = false;
78 }
79 }
80
81 mParent->mLogger->trace("[%] Looping has been stopped\n", mParent->mReader->getName());
82
83 } catch (const RuntimeException& e) {
84 dynamic_cast<ObservableLocalReaderAdapter*>(mParent->mReader)
85 ->getObservationExceptionHandler()
86 ->onReaderObservationError(
87 dynamic_cast<ObservableLocalReaderAdapter*>(mParent->mReader)->getPluginName(),
88 mParent->mReader->getName(),
89 std::make_shared<RuntimeException>(e));
90 }
91}
