44void AutonomousObservableLocalPluginAdapter::onReaderConnected(
45 const std::vector<std::shared_ptr<ReaderSpi>>& readers)
46{
47 Assert::getInstance().notEmpty(readers, "readers");
48 std::vector<std::string> notifyReaders;
49
50 for (const auto& readerSpi : readers) {
51 /* Add reader to plugin */
52 addReader(readerSpi);
53 notifyReaders.push_back(readerSpi->getName());
54 }
55
56 notifyObservers(
57 std::make_shared<PluginEventAdapter>(getName(),
58 notifyReaders,
59 PluginEvent::Type::READER_CONNECTED));
60}

62void AutonomousObservableLocalPluginAdapter::onReaderDisconnected(
63 const std::vector<std::string>& readerNames)
64{
65 Assert::getInstance().notEmpty(readerNames, "readerNames");
66 std::vector<std::string> notifyReaders;
67
68 for (const auto& readerName : readerNames) {
69 const std::shared_ptr<CardReader> reader = getReader(readerName);
70 if (reader == nullptr) {
71 mLogger->warn("[%] ObservableLocalPlugin => Impossible to remove reader, reader not " \
72 "found with name : %\n",
73 getName(),
74 readerName);
75 } else {
76 /* Unregister and remove reader */
77 std::dynamic_pointer_cast<LocalReaderAdapter>(reader)->doUnregister();
78 getReadersMap().erase(reader->getName());
79 mLogger->trace("[%] ObservableLocalPlugin => Remove reader '%' from readers list\n",
80 getName(),
81 reader->getName());
82
83 notifyReaders.push_back(readerName);
84 }
85 }
86
87 notifyObservers(
88 std::make_shared<PluginEventAdapter>(getName(),
89 notifyReaders,
90 PluginEvent::Type::READER_DISCONNECTED));
91}

93void AutonomousObservableLocalPluginAdapter::addReader(std::shared_ptr<ReaderSpi> readerSpi)
94{
95 std::shared_ptr<LocalReaderAdapter> reader = buildLocalReaderAdapter(readerSpi);
96 reader->doRegister();
97 getReadersMap().insert({reader->getName(), reader});
98
99 mLogger->trace("[%] ObservableLocalPlugin => Add reader '%' to readers list\n",
100 getName(),
101 readerSpi->getName());
102}
