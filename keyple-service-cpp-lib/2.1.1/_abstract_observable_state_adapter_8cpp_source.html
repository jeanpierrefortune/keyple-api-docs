50void AbstractObservableStateAdapter::onActivate()
51{
52 mLogger->trace("[%] onActivate => %\n", mReader->getName(), getMonitoringState());
53
54 /* Launch the monitoringJob if necessary */
55 if (mMonitoringJob != nullptr) {
56 if (mExecutorService == nullptr) {
57 throw IllegalStateException("ExecutorService must be set");
58 }
59
60 mMonitoringEvent = mExecutorService->submit(mMonitoringJob->getMonitoringJob(shared_from_this()));
61 }
62}
63
64void AbstractObservableStateAdapter::onDeactivate()
65{
66 mLogger->trace("[%] onDeactivate => %\n", mReader->getName(), getMonitoringState());
67
68 /* Cancel the monitoringJob if necessary */
69 if (mMonitoringEvent != nullptr && !mMonitoringEvent->isDone()) {
70 mMonitoringJob->stop();
71
72 const bool canceled = mMonitoringEvent->cancel(false);
73 const auto& id = *mMonitoringJob.get();
74 mLogger->trace("[%] onDeactivate => cancel monitoring job % by thread interruption %\n",
75 mReader->getName(),
76 typeid(id).name(),
77 canceled);
78 }
79
80 /* Make sure pointer is set to nullptr to force Job/Thread destruction */
81 mMonitoringEvent = nullptr;
82}
