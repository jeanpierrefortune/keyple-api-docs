62int CardSelectionManagerAdapter::prepareSelection(
63 const std::shared_ptr<CardSelection> cardSelection)
64{
65 Assert::getInstance().notNull(cardSelection, "cardSelection");
66
67 /* Keep the selection request */
68 auto selection = std::dynamic_pointer_cast<CardSelectionSpi>(cardSelection);
69 mCardSelections.push_back(selection);
70 std::shared_ptr<CardSelectionRequestSpi> selectionRequest = selection->getCardSelectionRequest();
71 mCardSelectionRequests.push_back(selectionRequest);
72
73 /* Return the selection index (starting at 0) */
74 return static_cast<int>(mCardSelections.size()) - 1;

82const std::shared_ptr<CardSelectionResult>
83 CardSelectionManagerAdapter::processCardSelectionScenario(std::shared_ptr<CardReader> reader)
84{
85 Assert::getInstance().notNull(reader, "reader");
86
87 /* Communicate with the card to make the actual selection */
88 std::vector<std::shared_ptr<CardSelectionResponseApi>> cardSelectionResponses;
89
90 try {
91 cardSelectionResponses =
92 std::dynamic_pointer_cast<AbstractReaderAdapter>(reader)
93 ->transmitCardSelectionRequests(mCardSelectionRequests,
94 mMultiSelectionProcessing,
95 mChannelControl);
96 } catch (const ReaderBrokenCommunicationException& e) {
97 throw ReaderCommunicationException(
98 e.getMessage(), std::make_shared<ReaderBrokenCommunicationException>(e));
99 } catch (const CardBrokenCommunicationException& e) {
100 throw CardCommunicationException(e.getMessage(),
101 std::make_shared<CardBrokenCommunicationException>(e));
102 }
103
104 /* Analyze the received responses */
105 return processCardSelectionResponses(cardSelectionResponses);
106 }

150const std::shared_ptr<CardSelectionResult>
151 CardSelectionManagerAdapter::processCardSelectionResponses(
152 const std::vector<std::shared_ptr<CardSelectionResponseApi>>& cardSelectionResponses) const
153{
154 Assert::getInstance().notEmpty(cardSelectionResponses, "cardSelectionResponses");
155
156 auto cardSelectionsResult = std::make_shared<CardSelectionResultAdapter>();
157
158 int index = 0;
159
160 /* Check card responses */
161 for (const auto& cardSelectionResponse : cardSelectionResponses) {
162 if (cardSelectionResponse->hasMatched()) {
163 /* Invoke the parse method defined by the card extension to retrieve the smart card */
164 std::shared_ptr<SmartCard> smartCard = nullptr;
165 try {
166 smartCard = std::dynamic_pointer_cast<SmartCard>(
167 mCardSelections[index]->parse(cardSelectionResponse));
168 } catch (const ParseException& e) {
169 throw InvalidCardResponseException(
170 "Error occurred while parsing the card response: " + e.getMessage(),
171 std::make_shared<ParseException>(e));
172 }
173
174 cardSelectionsResult->addSmartCard(index, smartCard);
175 }
176
177 index++;
178 }
179
180 return cardSelectionsResult;
181}
