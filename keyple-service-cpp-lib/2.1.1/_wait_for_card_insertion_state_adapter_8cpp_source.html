32void WaitForCardInsertionStateAdapter::onEvent(const InternalEvent event)
33{
34 mLogger->trace("[%] onInternalEvent => Event % received in currentState %\n",
35 getReader()->getName(),
36 event,
37 getMonitoringState());
38
39 /* Process InternalEvent */
40 switch (event) {
41 case InternalEvent::CARD_INSERTED:
42 {
43 /* Process default selection if any, return an event, can be null */
44 const std::shared_ptr<ReaderEvent> cardEvent = getReader()->processCardInserted();
45 if (cardEvent != nullptr) {
46 /* Switch internal state */
47 switchState(MonitoringState::WAIT_FOR_CARD_PROCESSING);
48 /* Notify the external observer of the event */
49 getReader()->notifyObservers(cardEvent);
50 } else {
51 /*
52 * If none event was sent to the application, back to card detection
53 * stay in the same state, however switch to WAIT_FOR_CARD_INSERTION to relaunch
54 * the monitoring job
55 */
56 mLogger->trace("[%] onInternalEvent => Inserted card hasn't matched\n",
57 getReader()->getName());
58
59 switchState(MonitoringState::WAIT_FOR_CARD_REMOVAL);
60 }
61 break;
62 }
63 case InternalEvent::STOP_DETECT:
64 switchState(MonitoringState::WAIT_FOR_START_DETECTION);
65 break;
66
67 case InternalEvent::CARD_REMOVED:
68 /* The card has been removed during default selection */
69 if (getReader()->getDetectionMode() == DetectionMode::REPEATING) {
70 switchState(MonitoringState::WAIT_FOR_CARD_INSERTION);
71 } else {
72 switchState(MonitoringState::WAIT_FOR_START_DETECTION);
73 }
74 break;
75
76 default:
77 mLogger->warn("[%] Ignore => Event % received in currentState %\n",
78 getReader()->getName(),
79 event,
80 getMonitoringState());
81 break;
82 }
83}
