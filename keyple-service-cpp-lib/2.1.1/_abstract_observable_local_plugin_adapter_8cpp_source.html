59void AbstractObservableLocalPluginAdapter::notifyObservers(const std::shared_ptr<PluginEvent> event)
60{
61 mLogger->debug("Plugin '%' notifies the plugin event '%' to % observer(s)\n",
62 getName(),
63 event->getType(),
64 countObservers());
65
66 for (const auto& observer : mObservationManager->getObservers()) {
67 notifyObserver(observer, event);
68 }
69}

71void AbstractObservableLocalPluginAdapter::notifyObserver(
72 std::shared_ptr<PluginObserverSpi> observer, const std::shared_ptr<PluginEvent> event)
73{
74 try {
75 observer->onPluginEvent(event);
76 } catch (const Exception& e) {
77 try {
78 mObservationManager->getObservationExceptionHandler()
79 ->onPluginObservationError(getName(),
80 std::make_shared<Exception>(e));
81 } catch (const Exception& e2) {
82 mLogger->error("Exception during notification: %\n", e2);
83 mLogger->error("Original cause: %\n", e);
84 }
85 }
86}

107void AbstractObservableLocalPluginAdapter::removeObserver(
108 const std::shared_ptr<PluginObserverSpi> observer)
109{
110 Assert::getInstance().notNull(observer, "observer");
111
112 const auto& observers = mObservationManager->getObservers();
113 const auto it = std::find(observers.begin(), observers.end(), observer);
114
115 if (it != observers.end()) {
116 mObservationManager->removeObserver(observer);
117 }
118}
