36void CardInsertionPassiveMonitoringJobAdapter::CardInsertionPassiveMonitoringJob::execute()
37{
38 try {
39 while (!isInterrupted()) {
40 try {
41 mParent->mReaderSpi->waitForCardInsertion();
42 mMonitoringState->onEvent(InternalEvent::CARD_INSERTED);
43 break;
44 } catch (const ReaderIOException& e) {
45 (void)e;
46 /* Just warn as it can be a disconnection of the reader */
47 mParent->mLogger->warn("[%] waitForCardPresent => Error while processing card " \
48 "insertion event\n",
49 mParent->getReader()->getName());
50 break;
51 } catch (const TaskCanceledException& e) {
52 (void)e;
53 break;
54 }
55 }
56 } catch (const RuntimeException& e) {
57 mParent->getReader()->getObservationExceptionHandler()
58 ->onReaderObservationError(mParent->getReader()->getPluginName(),
59 mParent->getReader()->getName(),
60 std::make_shared<RuntimeException>(e));
61 }
62}
