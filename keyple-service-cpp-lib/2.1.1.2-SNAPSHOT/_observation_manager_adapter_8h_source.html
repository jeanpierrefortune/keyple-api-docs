68 void addObserver(std::shared_ptr<T> observer)
69 {
70 mLogger->debug("% is adding the observer '%'\n",
71 mOwnerComponent,
72 observer != nullptr ? typeid(*observer.get()).name() : "null");

74 Assert::getInstance().notNull(observer, "observer");

95 mLogger->debug("% is removing the observer '%'\n",

98
99 const std::lock_guard<std::mutex> lock(mMonitor);
100
101 mObservers.erase(std::remove(mObservers.begin(), mObservers.end(), observer),
102 mObservers.end());
