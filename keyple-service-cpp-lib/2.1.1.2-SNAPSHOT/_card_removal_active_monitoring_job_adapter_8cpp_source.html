46void CardRemovalActiveMonitoringJobAdapter::CardRemovalActiveMonitoringJob::execute()
47{
48 try {
49 mParent->mLogger->debug("[%] Polling from isCardPresentPing\n",
50 mParent->getReader()->getName());
51
52 /* Re-init loop value to true */
53 mParent->mLoop = true;
54
55 while (mParent->mLoop) {
56 if (!mParent->getReader()->isCardPresentPing()) {
57 mParent->mLogger->debug("[%] the card stopped responding\n",
58 mParent->getReader()->getName());
59 mMonitoringState->onEvent(InternalEvent::CARD_REMOVED);
60 return;
61 }
62
63 mRetries++;
64
65 mParent->mLogger->trace("[%] Polling retries : %\n",
66 mParent->getReader()->getName(),
67 mRetries);
68
69 try {
70 /* Wait a bit */
71 Thread::sleep(mParent->mCycleDurationMillis);
72 } catch (const InterruptedException& ignored) {
73 (void)ignored;
74 /* Restore interrupted state... */
75 interrupt();
76 mParent->mLoop = false;
77 }
78 }
79
80 mParent->mLogger->debug("[%] Polling loop has been stopped\n",
81 mParent->getReader()->getName());
82 } catch (const RuntimeException& e) {
83 mParent->getReader()->getObservationExceptionHandler()
84 ->onReaderObservationError(mParent->getReader()->getPluginName(),
85 mParent->getReader()->getName(),
86 std::make_shared<RuntimeException>(e));
87 }
88}
